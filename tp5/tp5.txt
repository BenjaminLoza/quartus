begin
	Sum<= A xor B xor Cin;              
	Cout<= (A and B) or (A and Cin) or (B and Cin);  
end;

-----con procesos

architecture behaviour of SUMADOR_COMPLETO is
signal caso:std_logic_vector(2 downto 0);
begin
caso<=(A&B&Cin);
	process (caso)
	begin
        case caso is
            when "000" => 
                S <= '0';Cout <= '0';
            when "001" => 
                S <= '1';Cout <= '0';
            when "010" => 
                S <= '1';Cout <= '0';
            when "011" => 
                S <= '0';Cout <= '1';
            when "100" => 
                S <= '1';Cout <= '0';
            when "101" => 
                S <= '0';Cout <= '1';
            when "110" => 
                S <= '0';Cout <= '1';
            when "111" => 
                S <= '1';Cout <= '1';
        end case;
    end process;
end architecture;
